# TUI (Terminal User Interface) Applications
# For a terminal-first, keyboard-driven workflow
# Install with: pacman -S --needed - < packages/tui.txt

# System Monitoring and Management
btop              # Modern resource monitor (CPU, RAM, disk, network)
ncdu              # Disk usage analyzer with ncurses interface
iotop             # I/O monitoring
bandwhich         # Network bandwidth monitor per process

# File Management
yazi              # Modern file manager (Rust, async, image preview)
fd                # Fast find alternative (used by yazi)
ripgrep           # Fast grep alternative (rg)
fzf               # Fuzzy finder (essential for productivity)
bat               # Cat with syntax highlighting
eza               # Modern ls replacement with colors and icons
zoxide            # Smarter cd command (learns your habits)
ouch              # Archive extraction/compression tool

# Git
lazygit           # TUI for git operations
git-delta         # Syntax highlighter for git diffs

# Audio Control
pulsemixer        # TUI/CLI mixer for PulseAudio/PipeWire

tmux              # Terminal multiplexer (classic, stable)

newsboat          # RSS/Atom feed reader
